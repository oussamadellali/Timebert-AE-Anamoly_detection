# ---------- Positional Encoding ----------
class PositionalEncoding(nn.Module):
    def __init__(self, d_model, max_len=5000):
        super().__init__()
        pe = torch.zeros(max_len, d_model)
        position = torch.arange(0, max_len).unsqueeze(1)
        div_term = torch.exp(torch.arange(0, d_model, 2).float() * (-np.log(10000.0) / d_model))
        pe[:, 0::2] = torch.sin(position * div_term)
        pe[:, 1::2] = torch.cos(position * div_term)
        self.pe = pe.unsqueeze(0)

    def forward(self, x):
        return x + self.pe[:, :x.size(1)].to(x.device)

# ---------- Modules ----------
class TimeBERTEncoder(nn.Module):
    def __init__(self, input_dim, d_model=64, nhead=4, num_layers=3):
        super().__init__()
        self.input_proj = nn.Linear(input_dim, d_model)
        self.pos_encoder = PositionalEncoding(d_model)
        encoder_layer = nn.TransformerEncoderLayer(d_model, nhead)
        self.transformer = nn.TransformerEncoder(encoder_layer, num_layers)

    def forward(self, x):
        x = self.input_proj(x)
        x = self.pos_encoder(x)
        return self.transformer(x)

class MaskedDecoder(nn.Module):
    def __init__(self, d_model, output_dim):
        super().__init__()
        self.output_proj = nn.Linear(d_model, output_dim)

    def forward(self, x):
        return self.output_proj(x)

class ForecastDecoder(nn.Module):
    def __init__(self, d_model, output_dim):
        super().__init__()
        self.rnn = nn.GRU(d_model, d_model, batch_first=True)
        self.output_proj = nn.Linear(d_model, output_dim)

    def forward(self, x):
        x, _ = self.rnn(x)
        return self.output_proj(x)

class ClassifierHead(nn.Module):
    def __init__(self, d_model):
        super().__init__()
        self.classifier = nn.Sequential(
            nn.AdaptiveAvgPool1d(1),
            nn.Flatten(),
            nn.Linear(d_model, 64),
            nn.ReLU(),
            nn.Linear(64, 2)
        )

    def forward(self, x):
        x = x.transpose(1, 2)
        return self.classifier(x)

class HybridTimeBERTAE(nn.Module):
    def __init__(self, input_dim):
        super().__init__()
        self.encoder = TimeBERTEncoder(input_dim)
        self.masked_decoder = MaskedDecoder(64, input_dim)
        self.forecast_decoder = ForecastDecoder(64, input_dim)
        self.classifier_head = ClassifierHead(64)

    def forward(self, x, mask):
        encoded = self.encoder(x)
        masked_recon = self.masked_decoder(encoded)
        forecasted = self.forecast_decoder(encoded)
        classification = self.classifier_head(encoded)
        return masked_recon, forecasted, classification

def hybrid_loss(x, masked_recon, forecasted, mask, class_logits=None, class_labels=None, alpha=0.3):
    masked_loss = F.mse_loss(masked_recon[mask], x[mask])
    forecast_loss = F.mse_loss(forecasted, x)
    total_loss = masked_loss + forecast_loss
    if class_logits is not None and class_labels is not None:
        classification_loss = F.cross_entropy(class_logits, class_labels)
        total_loss += alpha * classification_loss
    return total_loss

# ---------- NYC Taxi Dataset Loader ----------
def load_nyc_taxi(filepath, seq_len=72):
    df = pd.read_csv(filepath, parse_dates=['timestamp'])
    anomaly_ranges = [
        ('2014-07-04 00:00:00', '2014-07-04 23:59:59'),
        ('2014-08-25 00:00:00', '2014-08-25 23:59:59'),
        ('2014-12-31 00:00:00', '2014-12-31 23:59:59')
    ]
    df['anomaly'] = 0
    for start, end in anomaly_ranges:
        df.loc[(df['timestamp'] >= start) & (df['timestamp'] <= end), 'anomaly'] = 1

    scaler = StandardScaler()
    df['value'] = scaler.fit_transform(df[['value']])

    X, y, timestamps = [], [], []
    for i in range(len(df) - seq_len):
        window = df.iloc[i:i + seq_len]
        label = int(window['anomaly'].any())
        X.append(window['value'].values)
        y.append(label)
        timestamps.append(window['timestamp'].values)

    X = np.array(X)[..., None]
    y = np.array(y)
    timestamps = np.array(timestamps)

    X_flat = X.reshape((X.shape[0], -1))
    X_resampled, y_resampled = ADASYN().fit_resample(X_flat, y)
    sample_indices = np.linspace(0, len(timestamps) - 1, num=len(X_resampled), dtype=int)
    resampled_timestamps = timestamps[sample_indices]
    X_resampled = X_resampled.reshape((-1, seq_len, 1))
    mask = np.random.rand(*X_resampled.shape) > 0.2

    dataset = TensorDataset(torch.tensor(X_resampled, dtype=torch.float32),
                            torch.tensor(y_resampled, dtype=torch.long),
                            torch.tensor(mask, dtype=torch.bool))
    return dataset, scaler, resampled_timestamps

        masked_recon, forecasted, logits = model(xb, mb)

        for xb, yb, mb in val_loader:

            val_preds.extend(torch.argmax(logits, dim=1).cpu().numpy())

    val_acc = accuracy_score(val_labels, val_preds)

    val_prec = precision_score(val_labels, val_preds)
    val_rec = recall_score(val_labels, val_preds)
